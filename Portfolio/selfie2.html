This project was about taking a selfie as reference and drawing your version of it through code with the program p5.js.

This is my code to make this drawing happen:

function setup() {
  createCanvas(600, 600);
background(128, 0, 255);
}

function draw ()  {
  //hair
 beginShape();
 fill(0,0,0);
 stroke(0,0,0);
 vertex(300,120);
 vertex(200,100);
 vertex(130,170);
 vertex(120,300);
 vertex(90,400);
 vertex(120,500);
 vertex(120,530);
 vertex(300,500);
 endShape(CLOSE);
 
 beginShape();
 vertex(300,120)
 vertex(400,105);
 vertex(470,145);
 vertex(517,190);
 vertex(500,323);
 vertex(517,367);
 vertex(485,400);
 vertex(513,450);
 vertex(470,525);
 vertex(300,500);
 endShape(CLOSE);
 //head
   stroke(255,255,255);
  fill(255, 242, 0);
  ellipse(300, 300, 300, 300);
    fill(0);
  strokeWeight(10)
  // upper eyebrows
  stroke(0,0,0);
  line(190, 250, 230, 240);
  line(230, 240, 270, 250);
  line(410, 250, 370, 240);
  line(370, 240, 330, 250);
  //eyebrows
  line(190, 300, 230, 290);
  line(230, 290, 270, 300);
  line(410, 300, 370, 290);
  line(370, 290, 330, 300);
  //eyelash
  line(190, 300, 180, 280);
  line(410, 300, 420, 280);
  //nose
  line(270, 360, 300, 310)
  //eyes
  circle(230,300,15);
  circle(370,300,15);
  //tuft
  beginShape();
  vertex(305,170);
  vertex(212,250);
  vertex(205,300);
  vertex(170,320);
  vertex(140,170);
  vertex(310,140);
  endShape(CLOSE);
  
  //neck
   beginShape();
   stroke(255,255,255);
   fill(255, 242, 0);
   vertex(260,450);
   vertex(260,475);
   vertex(340,475);
   vertex(340,450);
   endShape(CLOSE);
   //body
   stroke(255,255,255);
   fill(0,0,0);
     arc(300, 620, 300, 300, PI, 0, CHORD);
}
